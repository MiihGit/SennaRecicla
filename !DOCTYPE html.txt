        function togglePasswordVisibility(inputId, iconId) {
            const input = document.getElementById(inputId);
            const icon = document.getElementById(iconId);
            if (!input || !icon) return;

            if (input.type === "password") {
                input.type = "text";
                icon.classList.remove("fa-eye");
                icon.classList.add("fa-eye-slash");
            } else {
                input.type = "password";
                icon.classList.remove("fa-eye-slash");
                icon.classList.add("fa-eye");
            }
        }

        function validarSenhasMatch(id1, id2, msgId) {
            const s1 = document.getElementById(id1).value;
            const s2 = document.getElementById(id2).value;
            const msg = document.getElementById(msgId);
            
            if (s2 && s1 !== s2) {
                msg.style.display = 'block';
                document.getElementById(id2).style.border = '1px solid #cc0000';
            } else {
                msg.style.display = 'none';
                document.getElementById(id2).style.border = '1px solid #333';
            }
        }

        function fazerCadastroAluno() {
            const raNum = document.getElementById("cadRaNum").value.trim();
            const raDig = document.getElementById("cadRaDigito").value.trim();
            const raUF = document.getElementById("cadRaUF").value.trim().toUpperCase();
            
            const raCompleto = `${raNum}-${raDig} ${raUF}`;

            const numero = document.getElementById("cadNumero").value;
            const nome = document.getElementById("cadNome").value;
            const ano = document.getElementById("cadAno").value;
            const turma = document.getElementById("cadTurma").value;
            const senha = document.getElementById("cadSenhaAluno").value;
            const senhaConfirm = document.getElementById("cadSenhaAlunoConfirm").value;

            const msgErro = document.getElementById("erroCadAluno");
            msgErro.style.display = 'none'; // Reseta mensagem

            if(!raNum || !raDig || !raUF || !nome || !numero || !ano || !turma || !senha) { alert("Por favor, preencha todos os campos."); return; }
            if (senha !== senhaConfirm) { alert("As senhas não coincidem. Tente novamente."); return; }
            
            if(usuarios.find(u => u.ra === raCompleto)) { 
                msgErro.innerText = "Este RA já possui uma conta. Tente fazer login ou recuperar sua senha.";
                msgErro.style.display = 'block';
                return; 
            }
            if(usuarios.find(u => u.numero === numero && u.role === 'aluno')) { 
                msgErro.innerText = "Este telefone já está associado a uma conta. Tente fazer login.";
                msgErro.style.display = 'block';
                return; 
            }

            // Iniciar verificação
            verificationFlow = { type: 'signup', data: { ra: raCompleto, nome, numero, ano, turma, senha, role: 'aluno' } };
            // Aluno só tem telefone, então vai direto
            enviarCodigoDeVerificacao('sms');
        }

            if (senha !== senhaConfirm) { alert("As senhas não coincidem. Tente novamente."); return; }
            if (!email.includes('@') || !email.includes('.')) { alert("Por favor, insira um email válido."); return; }
            if (usuarios.find(u => u.email === email && u.role === 'professor')) {

            verificationFlow = { type: 'signup', data: { nome, email, telefone, senha, role: 'professor' } };

        function enviarCodigoDeVerificacao(method) {
            const data = verificationFlow.data;
            const code = Math.floor(100000 + Math.random() * 900000).toString();
            data.codigoVerificacao = code; // Adiciona o código aos dados pendentes
            
            let contactInfo = '';
            let message = '';

            // Esconde a tela de escolha (se estiver aberta)
            document.getElementById("contact-choice-auth").classList.add("hidden");

            if (method === 'email') {
                contactInfo = data.email;
                message = `Aguarde... Enviando código para: ${contactInfo}.`;

                // CHAMA O BACKEND PARA ENVIAR O EMAIL
                fetch('http://localhost:3000/enviar-email', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        email: contactInfo,
                        codigo: code,
                        tipo: 'Verificação de Conta'
                    })
                })
                .then(response => response.json())
                .then(result => {
                    if(result.sucesso) {
                        mostrarTelaInputCodigo(`Enviamos um código para o e-mail: ${contactInfo}. Verifique sua caixa de entrada (e spam).`, null);
                    } else {
                        alert("Erro ao enviar e-mail. Verifique se o servidor está rodando.");
                        console.error(result.erro);
                        // Fallback para simulação em caso de erro
                        mostrarTelaInputCodigo(`(ERRO NO ENVIO REAL) Código simulado: ${code}`, code);
                    }
                })
                .catch(err => {
                    console.error(err);
                    alert(`Não foi possível conectar ao servidor de e-mail. Verifique se o backend está rodando e veja o console do navegador (F12) para mais detalhes.\n\nErro: ${err.message}`);
                    mostrarTelaInputCodigo(`(ERRO DE CONEXÃO) Código simulado: ${code}`, code, true);
                });

            } else { // SMS (Continua simulado por enquanto)
                contactInfo = data.numero || data.telefone;
                message = `Enviamos um código via SMS para: ${contactInfo}.`;
                // Mostra a tela de inserir código com simulação
                mostrarTelaInputCodigo(message, code, true);
            }
        }

        function iniciarRecuperacao() {
            const identifier = document.getElementById("recoveryIdentifier").value.trim();
            if (!identifier) { return alert("Por favor, digite seu RA ou email."); }

            const user = usuarios.find(u => (u.ra === identifier && u.role === 'aluno') || (u.email === identifier && u.role === 'professor'));
            if (!user) { return alert("Usuário não encontrado. Verifique se digitou o RA (aluno) ou email (professor) corretamente."); }

            verificationFlow = { type: 'recovery', data: user };

            const hasEmail = !!user.email;
            const hasPhone = !!(user.numero || user.telefone);

            if (hasEmail && hasPhone) {
                mostrarTelaDeEscolhaDeContato();
            } else if (hasEmail) {
                enviarCodigoDeRecuperacao('email');
            } else if (hasPhone) {
                enviarCodigoDeRecuperacao('sms');
            } else {
                alert("Este usuário não possui um método de contato (email/telefone) para recuperação.");
                return;
            }
        }

        function enviarCodigoDeRecuperacao(method) {
            const user = verificationFlow.data;
            const code = Math.floor(100000 + Math.random() * 900000).toString();
            recoveryProcess = { user: user, code: code, identifier: user.ra || user.email };

            let contactMethod = (method === 'email') ? 'email' : 'telefone';
            let contact = (method === 'email') ? user.email : (user.numero || user.telefone);
            let maskedContact = (method === 'email') ? contact.slice(0, 2) + '...@' + contact.split('@')[1] : contact.slice(0, 4) + '... ' + contact.slice(-4);

            // Esconde a tela de escolha e a primeira etapa da recuperação
            document.getElementById("contact-choice-auth").classList.add("hidden");
            document.getElementById("recovery-step1").classList.add("hidden");
            document.getElementById("recovery-step2").classList.remove("hidden");

            if (method === 'email') {
                 document.getElementById("recovery-message").innerText = `Aguarde... Enviando código para o email...`;
                 
                 // CHAMA O BACKEND
                 fetch('http://localhost:3000/enviar-email', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        email: user.email,
                        codigo: code,
                        tipo: 'Recuperação de Senha'
                    })
                })
                .then(response => response.json())
                .then(result => {
                    if(result.sucesso) {
                        document.getElementById("recovery-message").innerText = `Enviamos um código para o email cadastrado: ${maskedContact}.`;
                    } else {
                        alert("Erro ao enviar e-mail real. Usando simulação.");
                        document.getElementById("recovery-message").innerText = `Erro no envio. Código simulado: ${code}`;
                        alert(`(SIMULAÇÃO) O seu código é: ${code}`);
                    }
                })
                .catch(err => {
                    alert(`Erro de conexão com o servidor. Verifique se o backend está rodando e veja o console do navegador (F12).\n\nErro: ${err.message}`);
                    document.getElementById("recovery-message").innerText = `Erro de conexão. Código simulado:`;
                    alert(`(SIMULAÇÃO) O seu código é: ${code}`);
                });

            } else {
                // SMS Simulado
                document.getElementById("recovery-message").innerText = `Enviamos um SMS para o número cadastrado: ${maskedContact}.`;
                alert(`(SIMULAÇÃO SMS) O seu código é: ${code}`);
            }

        function formatarDataParaBR(isoDate) { 
            if (!isoDate) return "";
            // Verifica se está no formato ISO (YYYY-MM-DD) antes de tentar converter
            if (isoDate.includes('-')) {
                const [year, month, day] = isoDate.substring(0, 10).split('-');
                return `${day}/${month}/${year}`;
            }
            return isoDate; // Retorna como está se for um formato antigo ou diferente
        }

        function renderizarAluno() {
            const tbody = document.querySelector("#tabela-aluno tbody");
            tbody.innerHTML = "";

            // Pega valores do filtro
            const dataInicio = document.getElementById("filtro-aluno-inicio").value;
            const dataFim = document.getElementById("filtro-aluno-fim").value;

            const meusRegistros = registros.filter(r => r.aluno === usuarioAtual.nome);
            let totalQtd = 0;

            const registrosFiltrados = meusRegistros.filter(r => {
                if (!dataInicio && !dataFim) return true; // Sem filtro
                const isAfterStart = dataInicio ? r.data >= dataInicio : true;
                const isBeforeEnd = dataFim ? r.data <= dataFim : true;
                return isAfterStart && isBeforeEnd;
            });

            registrosFiltrados.forEach(r => {
                const tr = document.createElement("tr");
                tr.innerHTML = `<td>${formatarDataParaBR(r.data)}</td><td>${r.material}</td><td>${r.qtd.toFixed(1).replace('.',',')} kg</td>`;
                tbody.appendChild(tr);
            });

            // O total geral do aluno não deve ser afetado pelo filtro da tabela
            totalQtd = meusRegistros.reduce((acc, r) => acc + parseFloat(r.qtd), 0);
            document.getElementById("aluno-total-qtd").innerText = totalQtd.toFixed(1).replace('.',',');
        }

                const user = usuarios.find(u => u.nome === r.aluno);
                if (!user) return false;
                const matchNome = fNome ? r.aluno.toLowerCase().includes(fNome.toLowerCase()) : true;
                const matchAno = fAno ? user.ano === fAno : true;
                const matchTurma = fTurma ? user.turma === fTurma : true;
                return matchNome && matchAno && matchTurma;
            });

            html += `<div class="${showAlunos}"><h2>Registros de Alunos</h2><table class="tabela-geral"><thead><tr><th>Data</th><th>Aluno (Turma)</th><th>Material</th><th>Qtd</th><th>Ações</th></tr></thead><tbody>`;
            registrosAlunosFiltrados.forEach(r => {
                const user = usuarios.find(u => u.nome === r.aluno);
                html += `<tr>
                    <td>${formatarDataParaBR(r.data)}</td>
                    <td>${r.aluno} <small>(${user ? user.ano : '?'} ${user ? user.turma : '?'})</small></td>
                    <td>${r.material}</td>
                    <td>${r.qtd.toFixed(1)}</td>
                    <td><button class="btn-edit" onclick="editarRegistro(${r.id})">Editar</button><button class="btn-delete" onclick="excluirRegistro(${r.id})">Excluir</button></td>
                </tr>`;
            });
            if (registrosAlunosFiltrados.length === 0) html += `<tr><td colspan="5" style="text-align:center;">${(fNome || fAno || fTurma) ? 'Nenhum resultado encontrado.' : 'Nenhum registro de aluno.'}</td></tr>`;
            html += '</tbody></table></div>';

            const registrosProfessoresFiltrados = registrosProfessores.filter(r => {
                return fNome ? r.aluno.toLowerCase().includes(fNome.toLowerCase()) : true;
            });


            html += `<div class="${showProf}"><h2>Registros de Professores</h2><table class="tabela-geral"><thead><tr><th>Data</th><th>Professor</th><th>Material</th><th>Qtd</th><th>Ações</th></tr></thead><tbody>`;
            registrosProfessoresFiltrados.forEach(r => {
                html += `<tr>
                    <td>${formatarDataParaBR(r.data)}</td>
                    <td>${r.aluno}</td>
                    <td>${r.material}</td>
                    <td>${r.qtd.toFixed(1)}</td>
                    <td><button class="btn-edit" onclick="editarRegistro(${r.id})">Editar</button><button class="btn-delete" onclick="excluirRegistro(${r.id})">Excluir</button></td>
                </tr>`;
            });
            if (registrosProfessoresFiltrados.length === 0) html += `<tr><td colspan="5" style="text-align:center;">${fNome ? 'Nenhum resultado encontrado.' : 'Nenhum registro de professor.'}</td></tr>`;
            html += '</tbody></table></div>';

            container.innerHTML = html;
        }

        function editarRegistro(id) {
            const registro = registros.find(r => r.id === id);
            if(registro) {
                let novaQtd = prompt("Nova quantidade para " + registro.material + ":", registro.qtd);
                if(novaQtd !== null && novaQtd.trim() !== "") {
                    registro.qtd = parseFloat(novaQtd);
                    registro.valor = (registro.qtd * precos[registro.material]).toFixed(2);
                    salvarDados();
                    if(usuarioAtual.role === 'admin') renderizarAdmin();
                    if(usuarioAtual.role === 'professor') renderizarProfessorView();
                    atualizarTotalGeral();
                    atualizarRanking();
                    atualizarRankingIndividual();
                }
            }
        }

        function excluirRegistro(id) {
            if(confirm("Tem certeza que deseja apagar este registro?")) {
                registros = registros.filter(r => r.id !== id);
                salvarDados();
                if(usuarioAtual.role === 'admin') renderizarAdmin();
                if(usuarioAtual.role === 'professor') renderizarProfessorView();
                atualizarTotalGeral();
                atualizarRanking();
                atualizarRankingIndividual();
            }
        }

        function salvarDados() {
            localStorage.setItem('sennaRegistros', JSON.stringify(registros));
        }

        function salvarUsuarios() {
            localStorage.setItem('sennaUsuarios', JSON.stringify(usuarios));
        }

        function atualizarRanking() {
            let turmas = {};
            registros.forEach(r => {
                let user = usuarios.find(u => u.nome === r.aluno && u.role === 'aluno');
                if (user) {
                    let nomeTurma = user.ano + ' ' + user.turma;
                    if (!turmas[nomeTurma]) turmas[nomeTurma] = 0;
                    turmas[nomeTurma] += parseFloat(r.qtd);
                }
            });
            let rankingArray = Object.keys(turmas).map(key => ({ turma: key, total: turmas[key] })).sort((a, b) => b.total - a.total).slice(0, 3);
            const container = document.getElementById("ranking-container");
            container.innerHTML = "";
            if (rankingArray.length === 0) { container.innerHTML = "<p>Nenhuma doação de aluno registrada ainda.</p>"; return; }
            rankingArray.forEach((item, index) => {
                let classeExtra = index === 0 ? "top1" : "";
                container.innerHTML += `<div class="rank-card ${classeExtra}"><span class="rank-pos">${index + 1}º</span><span class="rank-turma">${item.turma}</span><span class="rank-valor">${item.total.toFixed(1).replace('.', ',')} kg</span></div>`;
            });
        }

        function verificarSenhaAdmin() {
            const senhaAdmin = prompt("Para executar esta ação, digite a senha de segurança:");
            if (senhaAdmin === "12345") return true;
            alert("Senha incorreta. Ação cancelada.");
            return false;
        }

        function editarUsuario(id) {
            if (!verificarSenhaAdmin()) return;
            const userIndex = usuarios.findIndex(u => u.ra === id || u.email === id);
            if (userIndex === -1) { return alert("Usuário não encontrado."); }
            const user = usuarios[userIndex];
            if (user.role === 'aluno') {
                const nome = prompt("Novo nome:", user.nome); if (nome !== null && nome.trim() !== "") user.nome = nome;
                const ra = prompt("Novo RA:", user.ra); if (ra !== null && ra.trim() !== "" && ra !== user.ra) { if (usuarios.some(u => u.ra === ra)) { alert("RA já cadastrado. Alteração ignorada."); } else { user.ra = ra; } }
                const numero = prompt("Novo Telefone:", user.numero); if (numero !== null) user.numero = numero;
                const ano = prompt("Novo Ano:", user.ano); if (ano !== null) user.ano = ano;
                const turma = prompt("Nova Turma:", user.turma); if (turma !== null) user.turma = turma;
            } else if (user.role === 'professor') {
                const nome = prompt("Novo nome:", user.nome); if (nome !== null && nome.trim() !== "") user.nome = nome;
                const email = prompt("Novo Email:", user.email); if (email !== null && email.trim() !== "" && email !== user.email) { if (usuarios.some(u => u.email === email)) { alert("Email já cadastrado. Alteração ignorada."); } else { user.email = email; } }
                const telefone = prompt("Novo Telefone:", user.telefone); if (telefone !== null) user.telefone = telefone;
            }
            salvarUsuarios();
            renderizarGerenciamentoUsuarios();
            alert("Usuário atualizado!");
        }

        function redefinirSenha(id) {
            if (!verificarSenhaAdmin()) return;
            const user = usuarios.find(u => u.ra === id || u.email === id);
            if (!user) { return alert("Usuário não encontrado."); }
            const novaSenha = prompt(`Digite a nova senha para ${user.nome}:`);
            if (novaSenha !== null && novaSenha.trim() !== "") {
                user.senha = novaSenha;
                salvarUsuarios();
                alert("Senha do usuário atualizada com sucesso!");
            } else if (novaSenha !== null) {
                alert("Senha não pode ser vazia. Operação cancelada.");
            }
        }

        function excluirUsuario(id) {
            if (!verificarSenhaAdmin()) return;
            
            const userIndex = usuarios.findIndex(u => u.ra === id || u.email === id);
            if (userIndex === -1) { return alert("Erro: Usuário não encontrado na lista."); }

            const user = usuarios[userIndex];
            const idDisplay = user.ra || user.email || "Sem ID";
            if (confirm(`Tem certeza que deseja excluir permanentemente o usuário ${user.nome} (${idDisplay})? Esta ação não pode ser desfeita.`)) {
                usuarios.splice(userIndex, 1);
                salvarUsuarios();
                alert("Usuário excluído com sucesso.");
                renderizarGerenciamentoUsuarios();
            }
        }

        function atualizarRankingIndividual() {
            let pessoas = {};
            registros.forEach(r => {
                let nomePessoa = r.aluno;

                pessoas[nomePessoa] += parseFloat(r.qtd);
            });
            let rankingArray = Object.keys(pessoas).map(key => ({ nome: key, total: pessoas[key] })).sort((a, b) => b.total - a.total).slice(0, 5);
            const container = document.getElementById("ranking-individual-container");
            container.innerHTML = "";
            if (rankingArray.length === 0) { container.innerHTML = "<p>Nenhuma doação registrada ainda.</p>"; return; }
            rankingArray.forEach((item, index) => {

                container.innerHTML += `<div class="rank-card ${classeExtra}"><span class="rank-pos">${index + 1}º</span><span class="rank-turma">${item.nome}</span><span class="rank-valor">${item.total.toFixed(1).replace('.', ',')} kg</span></div>`;
            });
        }
